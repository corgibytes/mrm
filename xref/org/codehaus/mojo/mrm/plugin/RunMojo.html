19  import java.net.UnknownHostException;
20  
21  import org.apache.maven.plugin.MojoExecutionException;
22  import org.apache.maven.plugin.MojoFailureException;
23  import org.apache.maven.plugins.annotations.Mojo;
24  import org.apache.maven.plugins.annotations.Parameter;
25  import org.codehaus.plexus.util.FileUtils;
26  import org.codehaus.plexus.util.StringUtils;
27  
28  /**
29   * This goal is used in-situ on a Maven project to allow integration tests based on the Maven Invoker to use a custom
30   * <code>settings.xml</code> and still work behind a proxy.
31   *
32   * @author Stephen Connolly
33   *
34   * Starts a mock repository manager for manual testing.
35   */
36  @Mojo( name = "run", requiresProject = false, requiresDirectInvocation = true, threadSafe = true )
37  public class RunMojo
38      extends AbstractStartMojo
39  {
40      /**
41       * ServletPath for the settings.xml, so it can be downloaded.
42       */
43      @Parameter( property = "mrm.settingsServletPath", defaultValue = "settings-mrm.xml" )
44      private String settingsServletPath; 
45      
46      /**
47       * {@inheritDoc}
48       */
49      public void doExecute()
50          throws MojoExecutionException, MojoFailureException
51      {
52          if ( !session.getSettings().isInteractiveMode() )
53          {
54              throw new MojoExecutionException(
55                  "Cannot run a mock repository in batch mode (as there is no way to signal shutdown) "
56                      + "use mrm:start instead" );
57          }
58          FileSystemServer mrm = createFileSystemServer( createArtifactStore() );
59          getLog().info( "Starting Mock Repository Manager" );
60          mrm.ensureStarted();
61          String url = mrm.getUrl();
62          try
63          {
64              getLog().info( "Mock Repository Manager " + url + " is started." );
65              if ( StringUtils.isNotEmpty( settingsServletPath ) )
66              {
67                  String downloadUrl;
68                  try
69                  {
70                      downloadUrl = mrm.getRemoteUrl();
71                  }
72                  catch ( UnknownHostException e )
73                  {
74                      downloadUrl = mrm.getUrl();
75                  }
76                  
77                  String settings = FileUtils.filename( settingsServletPath );
78                  
79                  getLog().info( "To share this repository manager, let users download " + downloadUrl + "/" + settingsServletPath );
80                  getLog().info( "Maven should be started as 'mvn --settings " + settings +" [phase|goal]'" );
81              }
82              ConsoleScanner consoleScanner = new ConsoleScanner();
83              consoleScanner.start();
84              getLog().info( "Hit ENTER on the console to stop the Mock Repository Manager and continue the build." );
85              consoleScanner.waitForFinished();
86          }
87          catch ( InterruptedException e )
88          {
89              // ignore
90          }
91          finally
92          {
93              getLog().info( "Stopping Mock Repository Manager " + url );
94              mrm.finish();
95              try
96              {
97                  mrm.waitForFinished();
98                  getLog().info( "Mock Repository Manager " + url + " is stopped." );
99              }
100             catch ( InterruptedException e )
101             {
102                 // ignore
103             }
104         }
105     }
106     
107     /**
108      * {@inheritDoc}
109      */
110     @Override
111     protected String getSettingsServletPath()
112     {
113         return settingsServletPath;
114     }
115 
116 }
