19  import org.apache.maven.artifact.ArtifactUtils;
20  import org.apache.maven.plugin.MojoExecutionException;
21  import org.apache.maven.plugins.annotations.Parameter;
22  import org.codehaus.mojo.mrm.api.maven.ArtifactStore;
23  import org.codehaus.mojo.mrm.impl.digest.AutoDigestFileSystem;
24  import org.codehaus.mojo.mrm.impl.maven.ArtifactStoreFileSystem;
25  import org.codehaus.mojo.mrm.impl.maven.CompositeArtifactStore;
26  
27  import java.util.ArrayList;
28  import java.util.List;
29  
30  /**
31   * Common base class for the mojos that start a repository.
32   *
33   * @since 1.0
34   */
35  public abstract class AbstractStartMojo
36      extends AbstractMRMMojo
37  {
38      /**
39       * The port to serve the repository on. If not specified a random port will be used.
40       */
41      @Parameter( property = "mrm.port" )
42      private int port;
43  
44      /**
45       * The repositories to serve. When more than one repository is specified, a merged repository view
46       * of those will be used. If none specified then a proxy of the invoking Maven's repositories will
47       * be served.
48       */
49      @Parameter
50      private ArtifactStoreFactory[] repositories;
51  
52      /**
53       * Creates a file system server from an artifact store.
54       *
55       * @param artifactStore the artifact store to serve.
56       * @return the file system server.
57       */
58      protected FileSystemServer createFileSystemServer( ArtifactStore artifactStore )
59      {
60          return new FileSystemServer( ArtifactUtils.versionlessKey( project.getGroupId(), project.getArtifactId() ),
61                                       Math.max( 0, Math.min( port, 65535 ) ),
62                                       new AutoDigestFileSystem( new ArtifactStoreFileSystem( artifactStore ) ), 
63                                       getSettingsServletPath() );
64      }
65  
66      /**
67       * When set, this points to the to the location from where the settings file can be downloaded. 
68       * 
69       * @return the servlet path to the settings file of {@code null}
70       */
71      protected String getSettingsServletPath()
72      {
73          return null;
74      }
75  
76      /**
77       * Creates an artifact store from the {@link #repositories} configuration.
78       *
79       * @return an artifact store.
80       * @throws org.apache.maven.plugin.MojoExecutionException
81       *          if the configuration is invalid.
82       */
83      protected ArtifactStore createArtifactStore()
84          throws MojoExecutionException
85      {
86          if ( repositories == null )
87          {
88              getLog().info( "Configuring Mock Repository Manager to proxy through this Maven instance" );
89              return createProxyArtifactStore();
90          }
91          getLog().info( "Configuring Mock Repository Manager..." );
92          List<ArtifactStore> stores = new ArrayList<>();
93          if ( repositories == null || repositories.length == 0 )
94          {
95              repositories = new ArtifactStoreFactory[]{ new ProxyRepo() };
96          }
97          FactoryHelper helper = createFactoryHelper();
98          for ( ArtifactStoreFactory artifactStoreFactory : repositories )
99          {
100             if ( artifactStoreFactory instanceof FactoryHelperRequired )
101             {
102                 ( (FactoryHelperRequired) artifactStoreFactory ).setFactoryHelper( helper );
103             }
104             getLog().info( "  " + artifactStoreFactory.toString() );
105             stores.add( artifactStoreFactory.newInstance() );
106         }
107         ArtifactStore[] artifactStores = stores.toArray(new ArtifactStore[0]);
108         return artifactStores.length == 1 ? artifactStores[0] : new CompositeArtifactStore( artifactStores );
109     }
